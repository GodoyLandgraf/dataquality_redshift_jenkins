pipeline {
    agent any

    environment {
        // Define environment variables
        AWS_DEFAULT_REGION = 'us-east-1'
        USERNAME = credentials('redshift_username')
        PASSWORD = credentials('redshift_password')
        HOST = credentials('host')
        ROLE_ARN = credentials('role_arn')
    }

    stages {
        stage('Checkout') {
            steps {
                  git credentialsId: 'github', url: 'https://github.com/GodoyLandgraf/dataquality_test.git'
                checkout scm
            }
        }
        stage('AWS Configure') {
            steps {
                withCredentials([
                    [
                        $class: 'AmazonWebServicesCredentialsBinding',
                        credentialsId: 'aws-key',
                        accessKeyVariable: 'AWS_ACCESS_KEY_ID',
                        secretKeyVariable: 'AWS_SECRET_ACCESS_KEY'                    
                    ]
                ]) {
                    script {
                        sh 'aws configure set region ${AWS_DEFAULT_REGION}'
                    }
                }
            }
        }

            stage('SODA Scan Checks PRODUCT 2') {
                steps {
                    script {
                        echo "Test Connection:"
                        sh 'soda test-connection -d redshift_datasource -c configuration.yml'
                        def sodaScanOutput = null
                        echo "Running soda scan:"
                        try {
                            sodaScanOutput = sh(script: 'soda scan -d redshift_datasource -c configuration.yml checks_table2.yml', returnStdout: true).trim()
                            // Debugging messages
                            echo "Soda Scan Output:"
                            echo sodaScanOutput
                            echo "Soda scan completed"
                            echo "Soda Scan Success"
                        } catch (Exception e) {
                            catchError(buildResult: 'SUCCESS', stageResult: 'FAILURE'){
                            error("Soda scan failed") // This will mark the stage as failed
                        }
                        } finally {
                            // Save the output even if there was a failure
                            if (sodaScanOutput != null) {
                                def outputFile = "${WORKSPACE}/checks_redshift_table2_${BUILD_ID}"
                                writeFile file: outputFile, text: sodaScanOutput
                                sh "aws s3 cp ${outputFile} s3://athena-soda-storage/jenkins-storage/"
                            }
                        }
                    }
                }
            }
        stage('SODA Scan Checks PRODUCT 1') {
            steps {
                script {
                    // Ensure we are in the correct directory
                        sh 'pwd'
                        echo "Test Connection:"
                        sh 'soda test-connection -d redshift_datasource -c configuration.yml'
                        echo "Running soda scan:"
                        def sodaScanOutput = sh(script: 'soda scan -d redshift_datasource -c configuration.yml checks_table1.yml || true', returnStdout: true).trim()
                        // Debugging messages
                        echo "Soda Scan Output:"
                        echo sodaScanOutput
                        echo "Soda scan completed"
                        writeFile file: "${WORKSPACE}/checks_redshift_table1_${BUILD_ID}", text: sodaScanOutput
                }
            }
        }
    }
}
